### Document.ready

	<script type="text/javascript">
		$(document).ready(function() {
			// jQuery code goes here

### jQuery selectors

	// by element type
	$("a")

	// by id
	$("#container")

	// by class
	$(".article")

	// descendents
	$("#container p")
